<?php
    require_once("autoload.php");
    class Test extends Conexion
    {
        public function getUser()
        {
            $sql = "SELECT * FROM usuario";
            $stmt = $this->connect()->query($sql);
            while($row = $stmt->fetch())
            {
                echo $row['nombre'] . '<br>';
                echo "It's working";
            }echo "It's working";
        }


        public function getUserstmt($nombre, $telefono)
        {
            $sql = "SELECT * FROM usuario WHERE nombre = ? AND telefono = ?";
            $stmt = $this->connect()->prepare($sql);
            $stmt->execute([$nombre, $telefono]);
            $names = $stmt->fetchAll();
            foreach($names as $name)
            {
                echo $name['nombre'] . '<br>';
            }
            echo "It's working";
        }


        public function setUserstmt($nombre, $telefono, $email)
        {
            $sql = "INSERT INTO usuario(nombre, telefono, email) VALUES(?,?,?);";
            $stmt = $this->connect()->prepare($sql);
            $stmt->execute([$nombre, $telefono, $email]);
            $names = $stmt->fetchAll();
            foreach($names as $name)
            {
                echo $name['nombre'] . '<br>';
            }
            echo "It's working";
        }
    }// End of class test
?>
